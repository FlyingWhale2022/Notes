



```matlab
% 初始值
x0 = 1;
y0 = 1;

% 前后迭代误差
epsilon = 0.0001;

% 初始化迭代计数器
iteration = 1;

% 迭代
while true
    % 计算矩阵
    A = [2 * x0, 2 * y0; 2 * (x0 - 1), 2 * y0];
    B = [x0^2 + y0^2 - 1; (x0 - 1)^2 + y0^2 - 1];
    
    % 计算更新值
    delta = -inv(A) * B;
    x_new = x0 + delta(1);
    y_new = y0 + delta(2);
    
    % 计算前后迭代误差
    error = norm([x_new; y_new] - [x0; y0]);
    
    % 输出当前迭代结果，包括迭代编号
    fprintf('第 %d 次迭代 x: %.7f, y: %.7f, error: %.7f\n', iteration, x_new, y_new, error);
    
    % 检查是否满足停止条件
    if error < epsilon
        break;
    end
    
    % 更新迭代值和计数器
    x0 = x_new;
    y0 = y_new;
    iteration = iteration + 1;
end
```
